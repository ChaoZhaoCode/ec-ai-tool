📊 Google Sheets連携コード 面接用 Apps Script
▼

以下のコードをGoogle Sheetsのスクリプトエディタ（拡張機能 → Apps Script）に貼り付けると、スプレッドシートでも同じ機能が使えます。
A列:商品名 B列:仕様 C列:キーワード → D列:楽天文案 E列:Amazon文案 が自動生成されます。

STEP 01
スプレッドシート作成
Google Sheetsで新規シートを作成。1行目にヘッダーを設定。
STEP 02
スクリプト貼り付け
拡張機能→Apps Script→コードを貼り付けてAPIキーを設定。
STEP 03
実行・承認
初回実行時にGoogleの承認画面が出るので許可する。
STEP 04
一括生成
メニューに「AI文案生成」が追加され、ワンクリックで実行可能。
// ================================
// EC AI コピーライター - Google Apps Script
// 使い方: スプレッドシートのメニュー →「AI文案生成」
// ================================

const API_KEY = 'sk-proj-ここにOpenAIのAPIキーを入れる';
const MODEL = 'gpt-4o';

// メニューをスプレッドシートに追加する
function onOpen() {
  SpreadsheetApp.getUi()
    .createMenu('🤖 AI文案生成')
    .addItem('選択行を生成', 'generateSelected')
    .addItem('全行を一括生成', 'generateAll')
    .addToUi();
}

// 全行を一括生成（メインの関数）
function generateAll() {
  const sheet = SpreadsheetApp.getActiveSheet();
  const lastRow = sheet.getLastRow();
  
  // ヘッダー行をスキップして2行目から処理
  for (let i = 2; i <= lastRow; i++) {
    const name     = sheet.getRange(i, 1).getValue(); // A列: 商品名
    const spec     = sheet.getRange(i, 2).getValue(); // B列: 仕様
    const keywords = sheet.getRange(i, 3).getValue(); // C列: キーワード
    
    if (!name) continue; // 空行スキップ
    
    // D列: 楽天文案を生成
    sheet.getRange(i, 4).setValue(
      callOpenAI(buildPrompt(name, spec, keywords, 'rakuten'))
    );
    
    // E列: Amazon文案を生成
    sheet.getRange(i, 5).setValue(
      callOpenAI(buildPrompt(name, spec, keywords, 'amazon'))
    );

    // F列: 自社EC文案を生成
    sheet.getRange(i, 6).setValue(
      callOpenAI(buildPrompt(name, spec, keywords, 'general'))
    );
    
    // レート制限回避のため少し待機
    Utilities.sleep(1000);
  }
  
  SpreadsheetApp.getUi().alert('✅ 生成完了！全行の文案が生成されました。');
}

// プラットフォーム別プロンプトを組み立てる
function buildPrompt(name, spec, keywords, platform) {
  const styles = {
    rakuten: `楽天市場向け。感情・体験訴求型で600〜900字の自然な文章。
「〜で毎日が変わる」「大切な人へのギフトに」などの感情的表現を使い、
使用シーンを具体的に描写。キーワードを3〜5回自然に含める。`,
    amazon: `Amazon JP向け。以下の形式で厳密に出力すること：
【商品タイトル】（80文字以内）
【特徴1】〜【特徴5】（各60字以内の箇条書き）
【商品説明】（200字）
数値・スペック重視、客観的・情報的なトーン。`,
    general: `自社ECサイト向け。ブランド価値×機能説明のバランス型で500〜700字。
素材・こだわりを上品に訴求し、使用シーンと購買を後押しするクロージングを含める。`
  };
  
  return `以下の商品の商品説明文を日本語で作成してください。

商品名: ${name}
仕様・素材: ${spec}
SEOキーワード: ${keywords}

${styles[platform]}

商品説明文のみを出力してください。余計な説明は不要です。`;
}

// OpenAI APIを呼び出す
function callOpenAI(prompt) {
  const response = UrlFetchApp.fetch('https://api.openai.com/v1/chat/completions', {
    method: 'post',
    headers: {
      'Authorization': `Bearer ${API_KEY}`,
      'Content-Type': 'application/json'
    },
    payload: JSON.stringify({
      model: MODEL,
      max_tokens: 1200,
      temperature: 0.7,
      messages: [
        {
          role: 'system',
          content: 'あなたは日本のEC専門プロコピーライターです。各プラットフォームの特性を熟知し、SEOと購買率を両立した高品質な商品説明文を作成します。'
        },
        { role: 'user', content: prompt }
      ]
    }),
    muteHttpExceptions: true
  });
  
  const data = JSON.parse(response.getContentText());
  return data.choices?.[0]?.message?.content || 'エラー: 生成に失敗しました';
}
